Paso 3: Código y explicación

API REST: Usamos fetch y async/await para consumir el API.

// src/api.js
const API_URL = "https://api.ejemplo.com/clientes";

export async function getClientes() {
  const res = await fetch(API_URL);
  return await res.json();
}

export async function crearCliente(cliente) {
  const res = await fetch(API_URL, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(cliente),
  });
  return await res.json();
}

export async function actualizarCliente(id, cliente) {
  const res = await fetch(`${API_URL}/${id}`, {
    method: "PUT",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(cliente),
  });
  return await res.json();
}

export async function borrarCliente(id) {
  await fetch(`${API_URL}/${id}`, { method: "DELETE" });
}

Explicación:

  • Funciones async para cada operación CRUD.
  • Usa fetch y JSON para enviar/recibir datos.

Formulario de cliente: Permite crear o editar clientes.

// src/ClienteForm.jsx
import { useState, useEffect } from "react";

export default function ClienteForm({ onGuardar, clienteActual }) {
  const [nombre, setNombre] = useState("");
  const [email, setEmail] = useState("");

  useEffect(() => {
    if (clienteActual) {
      setNombre(clienteActual.nombre);
      setEmail(clienteActual.email);
    } else {
      setNombre("");
      setEmail("");
    }
  }, [clienteActual]);

  function handleSubmit(e) {
    e.preventDefault();
    onGuardar({ nombre, email });
    setNombre("");
    setEmail("");
  }

  return (
    <form className="bg-white rounded-xl shadow p-6 mb-6 flex flex-col gap-4" onSubmit={handleSubmit} >
      <h2 className="text-xl font-bold text-blue-700">{clienteActual ? "Editar Cliente" : "Nuevo Cliente"}</h2>
      <input className="border rounded px-3 py-2" value={nombre} onChange={e => setNombre(e.target.value)} placeholder="Nombre" required />
      <input className="border rounded px-3 py-2" value={email} onChange={e => setEmail(e.target.value)} placeholder="Email" type="email" required />
      <button className="bg-blue-600 text-white px-4 py-2 rounded hover:bg-blue-700" type="submit" >{clienteActual ? "Actualizar" : "Guardar"}</button>
    </form>
  );
}

Explicación:

  • Formulario controlado con useState y useEffect.
  • Permite crear o editar según el estado clienteActual.

Lista de clientes: Muestra los clientes y acciones CRUD.

// src/ClienteList.jsx
export default function ClienteList({ clientes, onEditar, onBorrar }) {
  return (
    <div className="bg-white rounded-xl shadow p-6">
      <h2 className="text-xl font-bold text-blue-700 mb-4">Lista de Clientes</h2>
      <table className="w-full text-left">
        <thead>
          <tr>
            <th className="py-2">Nombre</th>
            <th className="py-2">Email</th>
            <th className="py-2">Acciones</th>
          </tr>
        </thead>
        <tbody>
          {clientes.map(cliente => (
            <tr key={cliente.id} className="border-t">
              <td className="py-2">{cliente.nombre}</td>
              <td className="py-2">{cliente.email}</td>
              <td className="py-2 flex gap-2">
                <button className="bg-yellow-500 text-white px-3 py-1 rounded hover:bg-yellow-600" onClick={() => onEditar(cliente)} >Editar</button>
                <button className="bg-red-600 text-white px-3 py-1 rounded hover:bg-red-700" onClick={() => onBorrar(cliente.id)} >Borrar</button>
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

Explicación:

  • Muestra la lista de clientes en una tabla.
  • Botones para editar y borrar cada cliente.

Componente principal: Controla el estado global y las acciones CRUD.

// src/App.jsx
import { useState, useEffect } from "react";
import { getClientes, crearCliente, actualizarCliente, borrarCliente } from "./api";
import ClienteForm from "./ClienteForm";
import ClienteList from "./ClienteList";

export default function App() {
  const [clientes, setClientes] = useState([]);
  const [clienteActual, setClienteActual] = useState(null);

  async function cargar() {
    setClientes(await getClientes());
  }

  useEffect(() => {
    cargar();
  }, []);

  async function guardarCliente(data) {
    if (clienteActual) {
      await actualizarCliente(clienteActual.id, data);
    } else {
      await crearCliente(data);
    }
    setClienteActual(null);
    cargar();
  }

  async function borrar(id) {
    await borrarCliente(id);
    cargar();
  }

  function editar(cliente) {
    setClienteActual(cliente);
  }

  return (
    <div className="max-w-2xl mx-auto mt-10 space-y-8">
      <h1 className="text-3xl font-bold text-blue-800 mb-4">
        <i className="fa-solid fa-users"></i> CRUD Clientes React + API REST
      </h1>
      <ClienteForm onGuardar={guardarCliente} clienteActual={clienteActual} />
      <ClienteList clientes={clientes} onEditar={editar} onBorrar={borrar} />
    </div>
  );
}

Explicación:

  • Controla el estado global y las acciones CRUD.
  • Recarga la lista tras cada operación.
